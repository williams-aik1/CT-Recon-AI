import numpy as np
from .physics import compute_sinogram
from .data import add_gaussian_noise
from .reconstruction import fbp_reconstruction

def run_pipeline(image, angles, model):
    sinogram = compute_sinogram(image, angles)
    noisy = add_gaussian_noise(sinogram)
    recon = fbp_reconstruction(noisy, angles)

    enhanced = model.predict(recon[np.newaxis, ..., np.newaxis])[0, ..., 0]

    return recon, enhanced
