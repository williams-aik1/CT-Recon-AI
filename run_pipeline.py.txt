import numpy as np
import matplotlib.pyplot as plt
from skimage.data import shepp_logan_phantom
from skimage.transform import resize

from src.model import build_residual_model
from src.pipeline import run_pipeline
from src.evaluation import compute_metrics

# Generate phantom
phantom = resize(shepp_logan_phantom(), (128, 128))
angles = np.linspace(0, 180, 60)

# Build model
model = build_residual_model()

# Simulate training data (simple version)
X_train = []
y_train = []

for _ in range(50):
    X_train.append(phantom)
    y_train.append(phantom)

X_train = np.array(X_train)
y_train = np.array(y_train)

model.fit(
    X_train[..., np.newaxis],
    y_train[..., np.newaxis],
    epochs=5,
    batch_size=4
)

# Run pipeline
recon, enhanced = run_pipeline(phantom, angles, model)

# Metrics
mse1, psnr1, ssim1 = compute_metrics(phantom, recon)
mse2, psnr2, ssim2 = compute_metrics(phantom, enhanced)

print("Before AI:", mse1, psnr1, ssim1)
print("After AI:", mse2, psnr2, ssim2)

# Visualization
plt.figure(figsize=(12,4))

plt.subplot(1,3,1)
plt.title("Original")
plt.imshow(phantom, cmap='gray')

plt.subplot(1,3,2)
plt.title("Reconstruction")
plt.imshow(recon, cmap='gray')

plt.subplot(1,3,3)
plt.title("AI Enhanced")
plt.imshow(enhanced, cmap='gray')

plt.show()
